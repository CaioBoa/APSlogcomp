*Tradução para C*
{
    int x = 0;
    while (x != 15)
    {
        if ( x == 10 )
        {
            printf("igual a 10");
        }
        else if ( x < 10 )
        {
            printf("menor que 10");
        }
        else
        {
            printf("maior que 10");
        }
        x = x + 1;
    }
}
